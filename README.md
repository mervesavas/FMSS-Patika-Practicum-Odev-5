## Install

```bash
npm install --save coffeebutton
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'coffeebutton'
import 'coffeebutton/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
